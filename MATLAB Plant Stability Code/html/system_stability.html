Transfer Functions

b_theta = [123300, 0, 43983576];
a_theta = [2500, 62350, 2092275, 18544029, 0];
G_theta = tf(b_theta, a_theta);

b_alpha = [-123300, 379764];
a_alpha = [2500, 62350, 2092275, 18544029];
G_alpha = tf(b_alpha, a_alpha);

G_theta Routh Hurwitz

routh(a_theta);
